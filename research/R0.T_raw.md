// Middleware for tenant isolation
const tenantMiddleware = async (req, res, next) => {
  const farmId = req.headers['x-farm-id'] || req.query.farmId;
  
  if (!farmId) {
    return res.status(400).json({ error: 'Farm ID required' });
  }
  
  req.farmId = farmId;
  next();
};

// Apply to routes
app.use('/api/products', tenantMiddleware, productRoutes);

### Inventory Management Across Farms

```javascript
// Distributed inventory service
class InventoryService {
  async reserveStock(farmId, productId, quantity) {
    return await prisma.$transaction(async (tx) => {
      // Lock row for update
      const inventory = await tx.$queryRaw`
        SELECT * FROM "Inventory" 
        WHERE "farmId" = ${farmId} AND "productId" = ${productId}
        FOR UPDATE
      `;
      
      if (inventory.quantity < quantity) {
        throw new InsufficientStockError(
          `Only ${inventory.quantity} available for ${productId}`
        );
      }
      
      await tx.inventory.update({
        where: { id: inventory.id },
        data: { quantity: { decrement: quantity } }
      });
      
      return { reserved: true, remaining: inventory.quantity - quantity };
    });
  }
  
  async syncAcrossFarms(farmIds) {
    // For aggregated "multi-farm basket" checkout
    return await Promise.all(
      farmIds.map(farmId => this.getAvailableStock(farmId))
    );
  }
}
```

## 3. Shopify API Integration Best Practices for Inventory Management

### Shopify Store Integration Architecture

```javascript
// shopify-integration.service.js
const Shopify = require('shopify-api-node');

class ShopifyInventoryService {
  constructor(credentials) {
    this.shopify = new Shopify({
      shopName: credentials.shopName,
      apiKey: credentials.apiKey,
      password: credentials.password,
      apiVersion: '2024-01'
    });
  }
  
  // Webhook handler for inventory updates
  async handleInventoryUpdate(webhookPayload) {
    const { variant_id, available } = webhookPayload;
    
    // Sync to local farm inventory database
    await prisma.inventory.updateMany({
      where: { shopifyVariantId: variant_id },
      data: { quantity: available }
    });
    
    // Trigger low stock alerts
    if (available < 10) {
      await this.checkLowStockAlert(webhookPayload);
    }
  }
  
  // Bulk sync products from Shopify to local catalog
  async syncProducts(options = {}) {
    const products = await this.shopify.product.list({
      limit: 250,
      ...options
    });
    
    return await prisma.$transaction(async (tx) => {
      for (const product of products) {
        await tx.product.upsert({
          where: { shopifyProductId: product.id.toString() },
          create: {
            shopifyProductId: product.id.toString(),
            name: product.title,
            description: product.body_html,
            price: product.variants[0].price,
            shopifyVariantId: product.variants[0].id.toString(),
            farmId: options.farmId
          },
          update: {
            name: product.title,
            description: product.body_html,
            price: product.variants[0].price,
            inventory: {
              update: {
                quantity: product.variants[0].inventory_quantity
              }
            }
          }
        });
